	ma[x+1][y+2]=1;
	ma[x-1][y+2]=1;
	ma[x-2][y+1]=1;
	ma[x+2][y-1]=1;
	ma[x+1][y-2]=1;

	ma[x-2][y-1]=1;

		for(int j=0;j<=m;j++){
			if(i==0&&j==0) continue;

		}
	}
	cout<<f[n][m];
}
