#include<iostream>
#include<cstring>
typedef unsigned long long ull;
using namespace std;
int n,cnt,m;
int a[15],f[100000];
int main(){
	for(int i=1;i<=10;i++) cin>>a[i];
	cin>>n;
	/*
	f[i]表示i公里的最优解 
	*/ 
    memset(f,0x7f,sizeof(f));
	f[1]=a[1];
	for(int i=2;i<=n;i++){
		f[i]=f[i-1]+a[1];
		for(int j=1;j<=min(i,10);j++){
	 		f[i]=min(f[i],a[j]+f[i-j]);
		}
	}
	cout<<f[n];
}
