刷表法

代码：

#include<iostream>
using namespace std;
int n,cnt,m,f[100][100],x,y;
int dir[4][5]{{-2,1},{2,1},{-1,2},{1,2}};
int main(){
	cin>>n>>m>>x>>y;
    swap(n,m);
	f[0][0]=1;
	for(int j=0;j<=n;j++){
		for(int i=0;i<=m;i++){
			if(i==x||j==y){
				f[i][j]=0;
				continue;
			}
			for(int k=0;k<4;k++){
				int nx=i+dir[k][0],ny=j+dir[k][1];
				if(nx>=0&&nx<=m&&ny>=0&&ny<=n) f[nx][ny]+=f[i][j];
			}
		} 
	}
	cout<<f[m][n];
}
